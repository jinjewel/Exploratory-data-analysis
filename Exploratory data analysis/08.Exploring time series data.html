### 8장 시계열 자료의 탐색
# 시간이 지남에 따라 발생 또는 관찰되는 자료
# 특정 시점에서 관찰값은 자신의 과거 정보들에 의존하고 이들 간의 상관관계는 매우 높은 것이 일반적이다.
# s : 신호, a : 잡음, x : 시점 t에 관측된 값, x = s + a

# 시계열 자료 - 모의실험
set.seed(1234)
t <- seq(1:60)
s <- sin((2*pi*t)/12) + sin((2*pi*t)/30) 
a <- rnorm(60,0,1)
x <- s + a
x11()
par(mfrow=c(3,1))
plot(s, type="l", ylim=c(-3,3), xlab="", col="blue")
plot(a, type="l", ylim=c(-3,3), xlab="", col="orange")
plot(x, type="l", ylim=c(-3,3), xlab="", col="red")

x11()
par(mfrow=c(2,1))
plot(x, type="l", ylim=c(-3,3), lty="dotted", main="x & s", xlab="t")
lines(s)
smooth.x <- smooth(x)
plot(smooth.x, type="l", ylim=c(-3,3), main="x & hat(s)", xlab="t", ylab="", col="blue")
lines(x, lty="dotted")

# 시계열 평활 - smooting(3RS3R)
x <- c(2.1, 9.8, 19.5, 22.5, 16.6, 16.1, 18.5, -3.4, 8.9, -25.2, -14, -0.4)
smooth(x, kind="3")
## 3 Tukey smoother resulting from  smooth(x = x, kind = "3") 
##  used 1 iterations
##  [1]   2.1   9.8  19.5  19.5  16.6  16.6  16.1   8.9  -3.4 -14.0 -14.0 -14.0
smooth(x, kind="3R")
## 3R Tukey smoother resulting from  smooth(x = x, kind = "3R") 
##  used 1 iterations
##  [1]   2.1   9.8  19.5  19.5  16.6  16.6  16.1   8.9  -3.4 -14.0 -14.0 -14.0
smooth(x, kind="3RSS")
## 3RSS Tukey smoother resulting from  smooth(x = x, kind = "3RSS") 
##  used 2 iterations
##  [1]   2.1   9.8  19.5  16.6  16.6  16.6  16.1   8.9  -3.4 -14.0 -14.0 -14.0
smooth(x, kind="3RS3R")
## 3RS3R Tukey smoother resulting from  smooth(x = x, kind = "3RS3R") 
##  used 3 iterations
##  [1]   2.1   9.8  16.6  16.6  16.6  16.6  16.1   8.9  -3.4 -14.0 -14.0 -14.0
